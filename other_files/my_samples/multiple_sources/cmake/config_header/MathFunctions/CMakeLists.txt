option (USE_SQRT_TABLE "Use sqrt table autogenerated header" OFF)
if (USE_SQRT_TABLE)
    add_executable(make_sqrt_table make_sqrt_table.cpp)
    include_directories( ${CMAKE_CURRENT_BINARY_DIR} )
    add_custom_command (
        OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/sqrt_table.h
        COMMAND make_sqrt_table ${CMAKE_CURRENT_BINARY_DIR}/sqrt_table.h
        DEPENDS make_sqrt_table
    )
    set (EXTRA_FILES ${EXTRA_FILES} sqrt_table.h)
endif (USE_SQRT_TABLE)
